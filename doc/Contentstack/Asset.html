def initialize(attrs)
  if attrs.class == String
    @uid = attrs
  else
    attrs = attrs.symbolize_keys
    @uid = attrs[:uid]
    @content_type = attrs[:content_type]
    @filename = attrs[:filename]
    @file_size = attrs[:file_size]
    @tags = attrs[:tags]
    @url = attrs[:url]
  end

  self
end
